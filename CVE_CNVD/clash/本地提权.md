```
curl -X POST "http://127.0.0.1:33211/start_clash" -H "Content-Type: application/json" -d '{
    "core_type": "verge-mihomo",
    "bin_path": "/bin/bash",
    "config_dir": "1",
    "config_file": "/home/wall/test.sh",
    "log_file": "/tmp/clash.log"
  }'
```
